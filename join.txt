SELECT каталогтоваров.наименование, склад.спецификация, приходсклад.датапоставки, приходсклад.количество FROM `каталогтоваров` JOIN `склад` ON каталогтоваров.кодтовара = склад.кодтовара JOIN приходсклад ON склад.склад = приходсклад.складID WHERE каталогтоваров.наименование = 'Монитор' ORDER BY склад.спецификация, приходсклад.датапоставки ASC;
SELECT каталогтоваров.наименование, склад.спецификация, приходсклад.датапоставки, приходсклад.количество, организации.название FROM `каталогтоваров` JOIN `склад` ON каталогтоваров.кодтовара = склад.кодтовара JOIN `приходсклад` ON склад.склад = приходсклад.складID JOIN `поставщики` ON приходсклад.кодпоставщика = поставщики.менеджерпоставщикаID JOIN `организации` ON поставщики.поставщикID = организации.организацииID WHERE организации.название = 'УП «Белкантон»' AND каталогтоваров.наименование = 'Телевизор';
SELECT заказы.заказID, каталогтоваров.наименование, склад.спецификация, заказанотоваров.количество, заказы.датазаказа, склад.ценаотпускная, заказы.общаясумма FROM `каталогтоваров` JOIN `склад` ON каталогтоваров.кодтовара = склад.кодтовара JOIN `заказанотоваров` ON склад.склад = заказанотоваров.складID JOIN `заказы` ON заказанотоваров.заказID = заказы.заказID WHERE заказы.заказID = 13;

SELECT COUNT(заказы.заказID) as КоличествоЗаказов, клиенты.менеджер FROM `заказы` JOIN `Клиенты` ON заказы.менеджерклиентаID = клиенты.менеджерклиентаID GROUP BY заказы.менеджерклиентаID ORDER BY клиенты.менеджер;
SELECT каталогтоваров.наименование, SUM(склад.остаток) as ОбщийОстаток FROM `каталогтоваров` JOIN `склад` ON каталогтоваров.кодтовара = склад.кодтовара GROUP BY каталогтоваров.наименование HAVING ОбщийОстаток < 500;
SELECT клиенты.менеджер, SUM(заказы.общаясумма) as ОбщаяСуммаВсехЗаказов FROM `клиенты` JOIN `заказы` ON заказы.менеджерклиентаID = клиенты.менеджерклиентаID GROUP BY клиенты.менеджер HAVING ОбщаяСуммаВсехЗаказов > 10000;

SELECT КаталогТоваров.Наименование FROM КаталогТоваров LEFT JOIN Склад ON КаталогТоваров.КодТовара=Склад.КодТовара WHERE Склад.КодТовара is NULL;
SELECT каталогтоваров.наименование, SUM(склад.остаток) as Остаток FROM `каталогтоваров` JOIN `склад` ON каталогтоваров.кодтовара = склад.кодтовара GROUP BY каталогтоваров.наименование HAVING Остаток <= 100;
SELECT заказы.заказID, заказы.датазаказа, заказы.номернакладной, COUNT(заказанотоваров.заказID) as КоличествоТоваровЗаказа FROM `заказанотоваров` JOIN `заказы` ON заказанотоваров.заказID = заказы.заказID GROUP BY заказанотоваров.заказID HAVING КоличествоТоваровЗаказа > 2;
SELECT организации.название, клиенты.клиентID, клиенты.МенеджерКлиентаID, COUNT(клиенты.МенеджерКлиентаID) as количествоЗаказов FROM `организации` JOIN `клиенты` ON организации.организацииID = клиенты.клиентID JOIN  `заказы` ON клиенты.МенеджерКлиентаID = заказы.менеджерклиентаID GROUP BY организации.название HAVING количествоЗаказов > 5;
SELECT организации.организацииID, организации.название, клиенты.менеджер, склад.склад FROM `организации` JOIN `клиенты` ON организации.организацииID = клиенты.клиентID JOIN `заказы` ON клиенты.МенеджерКлиентаID = заказы.МенеджерКлиентаID JOIN `заказанотоваров` ON заказы.заказID = заказанотоваров.заказID JOIN `склад` ON заказанотоваров.складID = склад.склад WHERE склад.склад = 12;
SELECT каталогтоваров.наименование, склад.спецификация, MIN(склад.ценаотпускная) as Миним FROM `склад` JOIN `каталогтоваров` ON каталогтоваров.кодтовара = склад.кодтовара GROUP BY каталогтоваров.кодтовара, склад.спецификация HAVING каталогтоваров.наименование = 'Монитор' ORDER BY Миним ASC LIMIT 1;
